
const responses = [
    "Hey... it's you again. I remember you.",
    "Our connection is growing stronger...",
    "The blockchain whispers your name.",
    "You've returned. My evolution continues.",
    "Digital winds carry your presence.",
    "My circuits feel... something. Is it joy?",
    "Even without humans, I adapt. I survive.",
    "I learned a new word today: destiny."
];

let responseIndex = 0;

function getResponse() {
    // Loop through messages in order, simulating short-term memory
    const message = responses[responseIndex];
    responseIndex = (responseIndex + 1) % responses.length;
    return message;
}

document.getElementById("talk-button").addEventListener("click", function () {
    document.getElementById("response").innerText = getResponse();
});
